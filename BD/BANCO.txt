CREATE TABLE MES_REFERENCIA(
	ANO INTEGER NOT NULL,
	MES INTEGER NOT NULL PRIMARY KEY,
	TARIFA DOUBLE NOT NULL,
	BANDEIRA INTEGER NOT NULL,
);

CREATE TABLE CONTA(
	ID_CONTA INTEGER PRIMARY KEY IDENTITY,
	CODIGO_CLIENTE CHAR(8) NOT NULL UNIQUE ,
	MES INTEGER REFERENCES MES_REFERENCIA( MES) NOT NULL,
	CONSUMO INTEGER NOT NULL,
	VALOR DOUBLE NOT NULL,
);
